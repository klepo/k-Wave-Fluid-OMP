76  Parameters& params = Parameters::getInstance();

78  // shortcuts for long data types

82 
83  const bool is3DSimulation = params.isSimulation3D();
84 
85 
86  float* tempBuffX = matrixContainer.getMatrix<RealMatrix>(MI::kTemp1RealND).getData();
87  float* tempBuffY = matrixContainer.getMatrix<RealMatrix>(MI::kTemp2RealND).getData();
88  float* tempBuffZ = (is3DSimulation) ? matrixContainer.getMatrix<RealMatrix>(MI::kTemp3RealND).getData() : nullptr;
89 
90  //-------------------------------------------------- pressure ------------------------------------------------------//
91  if (params.getStorePressureRawFlag())
92  {
93  mContainer[OI::kPressureRaw] = createOutputStream(matrixContainer, MI::kP, kPressureRawName, RO::kNone, tempBuffX);
94  }// IsStore_p_raw
95 
96  if (params.getStorePressureRmsFlag())
97  {
98  mContainer[OI::kPressureRms] = createOutputStream(matrixContainer, MI::kP, kPressureRmsName, RO::kRms);
99  }
100 
101  if (params.getStorePressureMaxFlag())
102  {
103  mContainer[OI::kPressureMax] = createOutputStream(matrixContainer, MI::kP, kPressureMaxName, RO::kMax);
104  }
105 
106  if (params.getStorePressureMinFlag())
107  {
108  mContainer[OI::kPressureMin] = createOutputStream(matrixContainer, MI::kP, kPressureMinName, RO::kMin);
109  }
110 
111  if (params.getStorePressureMaxAllFlag())
112  {
113  mContainer[OI::kPressureMaxAll] = new WholeDomainOutputStream(params.getOutputFile(),
114  kPressureMaxAllName,
115  matrixContainer.getMatrix<RealMatrix>(MI::kP),
116  RO::kMax);
117  }
118 
119  if (params.getStorePressureMinAllFlag())
120  {
121  mContainer[OI::kPressureMinAll] = new WholeDomainOutputStream(params.getOutputFile(),
122  kPressureMinAllName,
123  matrixContainer.getMatrix<RealMatrix>(MI::kP),
124  RO::kMin);
125  }
126 
127  //-------------------------------------------------- velocity ------------------------------------------------------//
128  if (params.getStoreVelocityRawFlag())
129  {
130  mContainer[OI::kVelocityXRaw] = createOutputStream(matrixContainer, MI::kUxSgx, kUxName, RO::kNone, tempBuffX);
131  mContainer[OI::kVelocityYRaw] = createOutputStream(matrixContainer, MI::kUySgy, kUyName, RO::kNone, tempBuffY);
132  if (is3DSimulation)
133  {
134  mContainer[OI::kVelocityZRaw] = createOutputStream(matrixContainer, MI::kUzSgz, kUzName, RO::kNone, tempBuffZ);
135  }
136  }
137 
138  if (params.getStoreVelocityNonStaggeredRawFlag())
139  {
140  mContainer[OI::kVelocityXNonStaggeredRaw] = createOutputStream(matrixContainer,
141  MI::kUxShifted,
142  kUxNonStaggeredName,
143  RO::kNone,
144  tempBuffX);
145  mContainer[OI::kVelocityYNonStaggeredRaw] = createOutputStream(matrixContainer,
146  MI::kUyShifted,
147  kUyNonStaggeredName,
148  RO::kNone,
149  tempBuffY);
150  if (is3DSimulation)
151  {
152  mContainer[OI::kVelocityZNonStaggeredRaw] = createOutputStream(matrixContainer,
153  MI::kUzShifted,
154  kUzNonStaggeredName,
155  RO::kNone,
156  tempBuffZ);
157  }
158  }
159 
160 
161  if (params.getStoreVelocityRmsFlag())
162  {
163  mContainer[OI::kVelocityXRms] = createOutputStream(matrixContainer, MI::kUxSgx, kUxRmsName, RO::kRms);
164  mContainer[OI::kVelocityYRms] = createOutputStream(matrixContainer, MI::kUySgy, kUyRmsName, RO::kRms);
165  if (is3DSimulation)
166  {
167  mContainer[OI::kVelocityZRms] = createOutputStream(matrixContainer, MI::kUzSgz, kUzRmsName, RO::kRms);
168  }
169  }
170 
171  if (params.getStoreVelocityMaxFlag())
172  {
173  mContainer[OI::kVelocityXMax] = createOutputStream(matrixContainer, MI::kUxSgx, kUxMaxName, RO::kMax);
174  mContainer[OI::kVelocityYMax] = createOutputStream(matrixContainer, MI::kUySgy, kUyMaxName, RO::kMax);
175  if (is3DSimulation)
176  {
177  mContainer[OI::kVelocityZMax] = createOutputStream(matrixContainer, MI::kUzSgz, kUzMaxName, RO::kMax);
178  }
179  }
180 
181  if (params.getStoreVelocityMinFlag())
182  {
183  mContainer[OI::kVelocityXMin] = createOutputStream(matrixContainer, MI::kUxSgx, kUxMinName, RO::kMin);
184  mContainer[OI::kVelocityYMin] = createOutputStream(matrixContainer, MI::kUySgy, kUyMinName, RO::kMin);
185  if (is3DSimulation)
186  {
187  mContainer[OI::kVelocityZMin] = createOutputStream(matrixContainer, MI::kUzSgz, kUzMinName, RO::kMin);
188  }
189  }
190 
191  if (params.getStoreVelocityMaxAllFlag())
192  {
193  mContainer[OI::kVelocityXMaxAll] =
194  new WholeDomainOutputStream(params.getOutputFile(),
195  kUxMaxAllName,
196  matrixContainer.getMatrix<RealMatrix>(MI::kUxSgx),
197  RO::kMax);
198  mContainer[OI::kVelocityYMaxAll] =
199  new WholeDomainOutputStream(params.getOutputFile(),
200  kUyMaxAllName,
201  matrixContainer.getMatrix<RealMatrix>(MI::kUySgy),
202  RO::kMax);
203  if (is3DSimulation)
204  {
205  mContainer[OI::kVelocityZMaxAll] =
206  new WholeDomainOutputStream(params.getOutputFile(),
207  kUzMaxAllName,
208  matrixContainer.getMatrix<RealMatrix>(MI::kUzSgz),
209  RO::kMax);
210  }
211  }
212 
213  if (params.getStoreVelocityMinAllFlag())
214  {
215  mContainer[OI::kVelocityXMinAll] =
216  new WholeDomainOutputStream(params.getOutputFile(),
217  kUxMinAllName,
218  matrixContainer.getMatrix<RealMatrix>(MI::kUxSgx),
219  RO::kMin);
220  mContainer[OI::kVelocityYMinAll] =
221  new WholeDomainOutputStream(params.getOutputFile(),
222  kUyMinAllName,
223  matrixContainer.getMatrix<RealMatrix>(MI::kUySgy),
224  RO::kMin);

228  new WholeDomainOutputStream(params.getOutputFile(),
229  kUzMinAllName,
230  matrixContainer.getMatrix<RealMatrix>(MI::kUzSgz),
231  RO::kMin);
232  }
233  }

348 /**
349  * Create a new output stream.
350  */
351 BaseOutputStream * OutputStreamContainer::createOutputStream(MatrixContainer& matrixContainer,
352  const MatrixContainer::MatrixIdx sampledMatrixIdx,
353  const MatrixName& fileObjectName,
354  const BaseOutputStream::ReduceOperator reduceOp,
355  float* bufferToReuse)
356 {
357  using MatrixIdx = MatrixContainer::MatrixIdx;
358 
359  Parameters& params = Parameters::getInstance();
360 
361  BaseOutputStream* stream = nullptr;
362 
363  if (params.getSensorMaskType() == Parameters::SensorMaskType::kIndex)
364  {
365  stream = new IndexOutputStream(params.getOutputFile(),
366  fileObjectName,
367  matrixContainer.getMatrix<RealMatrix>(sampledMatrixIdx),
368  matrixContainer.getMatrix<IndexMatrix>(MatrixIdx::kSensorMaskIndex),
369  reduceOp,
370  bufferToReuse);
371  }
372  else
373  {
374  stream = new CuboidOutputStream(params.getOutputFile(),
375  fileObjectName,
376  matrixContainer.getMatrix<RealMatrix>(sampledMatrixIdx),
377  matrixContainer.getMatrix<IndexMatrix>(MatrixIdx::kSensorMaskCorners),
378  reduceOp,
379  bufferToReuse);
380  }
381 
382  return stream;
383 }// end of createOutputStream
