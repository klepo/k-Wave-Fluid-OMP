566  }
567

569  {
570  getSensorMaskCorners().recomputeIndicesToCPP();
571  }

580  }

585  }
586

588  {
589  getPressureSourceIndex().recomputeIndicesToCPP();
590  }
591 
592

594  if (!mParameters.getRho0ScalarFlag())
595  { // non-uniform grid cannot be pre-calculated :-(

598  {
599  generateInitialDenisty<simulationDimension>();
600  }

604  getDtRho0Sgy().scalarDividedBy(mParameters.getDt());

607  getDtRho0Sgz().scalarDividedBy(mParameters.getDt());
608  }
609  }
610  }
611

614  {

617  }
618  else

621  }
622

625  (mParameters.getPressureSourceMode() == Parameters::SourceMode::kAdditive)) &&
626  (mParameters.getPressureSourceFlag() ||
627  mParameters.getVelocityXSourceFlag() ||
628  mParameters.getVelocityYSourceFlag() ||
629  mParameters.getVelocityZSourceFlag()))
630  {
631  generateSourceKappa();
632  }
633

635  computeC2();
636 
637  Logger::log(Logger::LogLevel::kBasic, kOutFmtDone);

641 /**
642  * Compute the main time loop of KSpaceFirstOrder solver.
643  */
644 template<Parameters::SimulationDimension simulationDimension>
645 void KSpaceFirstOrderSolver::computeMainLoop()
646 {
647  mActPercent = 0;
648  // set ActPercent to correspond the time index after recovery
649  if (mParameters.getTimeIndex() > 0)
650  {
651  mActPercent = (100 * mParameters.getTimeIndex()) / mParameters.getNt();
652  }
653 
654  // Progress header
655  Logger::log(Logger::LogLevel::kBasic,kOutFmtSimulationHeader);
656 
657  mIterationTime.start();
658 
659  while ((mParameters.getTimeIndex() < mParameters.getNt()) && (!isTimeToCheckpoint()))
660  {
661  const size_t timeIndex = mParameters.getTimeIndex();
662 
663  // compute velocity
664  computeVelocity<simulationDimension>();
665  // add in the velocity source term
666  addVelocitySource();
667 
668  // add in the transducer source term (t = t1) to ux
669  if (mParameters.getTransducerSourceFlag() > timeIndex)
670  {
671  // transducer source is added only to the x component of the particle velocity
672  addTransducerSource();
673  }
674 
675  // compute gradient of velocity
676  computeVelocityGradient<simulationDimension>();
677 
678  if (mParameters.getNonLinearFlag())
679  {
680  computeDensityNonliner<simulationDimension>();
681  }
682  else
683  {
684  computeDensityLinear<simulationDimension>();
685  }
686 
687 
688  // add in the source pressure term
689  addPressureSource<simulationDimension>();
690 
691  if (mParameters.getNonLinearFlag())
692  {
693  computePressureNonlinear<simulationDimension>();
694  }
695  else
696  {
697  computePressureLinear<simulationDimension>();
698  }
699 
700  // calculate initial pressure
701  if ((timeIndex == 0) && (mParameters.getInitialPressureSourceFlag() == 1))
702  {
703  addInitialPressureSource<simulationDimension>();
704  }
705 
706  storeSensorData();
707  printStatistics();
708  mParameters.incrementTimeIndex();
709  }// time loop
710 }// end of computeMainLoop

936  }
937

993
