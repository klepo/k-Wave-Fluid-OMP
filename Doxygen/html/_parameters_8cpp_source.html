102 void Parameters::init(int argc, char** argv)
103 {
104  mCommandLineParameters.parseCommandLine(argc, argv);
105 
106  if (getGitHash() != "")
107  {
108  Logger::log(Logger::LogLevel::kFull, kOutFmtGitHashLeft, getGitHash().c_str());
109  Logger::log(Logger::LogLevel::kFull, kOutFmtSeparator);
110  }
111  if (mCommandLineParameters.isPrintVersionOnly())
112  {
113  return;
114  }
115 
116  Logger::log(Logger::LogLevel::kBasic, kOutFmtReadingConfiguration);
117  readScalarsFromInputFile();
118 
119  if (mCommandLineParameters.isBenchmarkEnabled())
120  {
121  mNt = mCommandLineParameters.getBenchmarkTimeStepsCount();
122  }
123 
124  if ((mCommandLineParameters.getSamplingStartTimeIndex() >= mNt) ||
125  (mCommandLineParameters.getSamplingStartTimeIndex() < 0))
126  {
127  throw std::invalid_argument(Logger::formatMessage(kErrFmtIllegalSamplingStartTimeStep, 1l, mNt));
128  }
129 
130  Logger::log(Logger::LogLevel::kBasic, kOutFmtDone);
131 }// end of parseCommandLine

170 /**
171  * Read scalar values from the input HDF5 file.
172  */
173 void Parameters::readScalarsFromInputFile()
174 {
175  DimensionSizes scalarSizes(1, 1, 1);
176 
177  if (!mInputFile.isOpen())
178  {
179  // Open file -- exceptions handled in main
180  mInputFile.open(mCommandLineParameters.getInputFileName());
181  }
182 
183  mFileHeader.readHeaderFromInputFile(mInputFile);
184 
185  // check file type
186  if (mFileHeader.getFileType() != Hdf5FileHeader::FileType::kInput)
187  {
188  throw ios::failure(Logger::formatMessage(kErrFmtBadInputFileFormat, getInputFileName().c_str()));
189  }
190 
191  // check version
192  if (!mFileHeader.checkMajorFileVersion())
193  {
194  throw ios::failure(Logger::formatMessage(kErrFmtBadMajorFileVersion,
195  getInputFileName().c_str(),
196  mFileHeader.getFileMajorVersion().c_str()));
197  }
198 
199  if (!mFileHeader.checkMinorFileVersion())
200  {
201  throw ios::failure(Logger::formatMessage(kErrFmtBadMinorFileVersion,
202  getInputFileName().c_str(),
203  mFileHeader.getFileMinorVersion().c_str()));
204  }
205 
206  const hid_t rootGroup = mInputFile.getRootGroup();
207 
208  // read dimension sizes
209  size_t x, y, z;
210  mInputFile.readScalarValue(rootGroup, kNxName, x);
211  mInputFile.readScalarValue(rootGroup, kNyName, y);
212  mInputFile.readScalarValue(rootGroup, kNzName, z);
213 
214  mFullDimensionSizes = DimensionSizes(x, y, z);
215  mReducedDimensionSizes = DimensionSizes(((x / 2) + 1), y, z);
216 
217  mInputFile.readScalarValue(rootGroup, kNtName, mNt);
218 
219  mInputFile.readScalarValue(rootGroup, kDtName, mDt);
220  mInputFile.readScalarValue(rootGroup, kDxName, mDx);
221  mInputFile.readScalarValue(rootGroup, kDyName, mDy);
222  if (isSimulation3D())
223  {
224  mInputFile.readScalarValue(rootGroup, kDzName, mDz);
225  }
226 
227  mInputFile.readScalarValue(rootGroup, kCRefName, mCRef);
228 
229  mInputFile.readScalarValue(rootGroup, kPmlXSizeName, mPmlXSize);
230  mInputFile.readScalarValue(rootGroup, kPmlYSizeName, mPmlYSize);
231  if (isSimulation3D())
232  {
233  mInputFile.readScalarValue(rootGroup, kPmlZSizeName, mPmlZSize);
234  }
235 
236  mInputFile.readScalarValue(rootGroup, kPmlXAlphaName, mPmlXAlpha);
237  mInputFile.readScalarValue(rootGroup, kPmlYAlphaName, mPmlYAlpha);
238  if (isSimulation3D())
239  {
240  mInputFile.readScalarValue(rootGroup, kPmlZAlphaName, mPmlZAlpha);
241  }
242 
243  // if the file is of version 1.0, there must be a sensor mask index (backward compatibility)
244  if (mFileHeader.getFileVersion() == Hdf5FileHeader::FileVersion::kVersion10)
245  {
246  mSensorMaskIndexSize = mInputFile.getDatasetSize(rootGroup, kSensorMaskIndexName);
247 
248  //if -u_non_staggered_raw enabled, throw an error - not supported
249  if (getStoreVelocityNonStaggeredRawFlag())
250  {
251  throw ios::failure(kErrFmtNonStaggeredVelocityNotSupportedFileVersion);
252  }
253  }// version 1.0
254 
255  // This is the current version 1.1
256  if (mFileHeader.getFileVersion() == Hdf5FileHeader::FileVersion::kVersion11)
257  {
258 
259  // read sensor mask type as a size_t value to enum
260  size_t sensorMaskTypeNumericValue = 0;
261  mInputFile.readScalarValue(rootGroup, kSensorMaskTypeName, sensorMaskTypeNumericValue);
262 
263  // convert the size_t value to enum
264  switch (sensorMaskTypeNumericValue)
265  {
266  case 0: mSensorMaskType = SensorMaskType::kIndex;
267  break;
268  case 1: mSensorMaskType = SensorMaskType::kCorners;
269  break;
270  default:
271  {
272  throw ios::failure(kErrFmtBadSensorMaskType);
273  break;
274  }
275  }//case
276 
277  // read the input mask size
278  switch (mSensorMaskType)
279  {
280  case SensorMaskType::kIndex:
281  {
282  mSensorMaskIndexSize = mInputFile.getDatasetSize(rootGroup, kSensorMaskIndexName);
283  break;
284  }
285  case SensorMaskType::kCorners:
286  {
287  // mask dimensions are [6, N, 1] - I want to know N
288  mSensorMaskCornersSize = mInputFile.getDatasetDimensionSizes(rootGroup, kSensorMaskCornersName).ny;
289  break;
290  }
291  }// switch
292  }// version 1.1
293 
294 
295  // flags.
296  mInputFile.readScalarValue(rootGroup, kPressureSourceFlagName, mPressureSourceFlag);
297  mInputFile.readScalarValue(rootGroup, kInitialPressureSourceFlagName,mInitialPressureSourceFlag);
298 
299  mInputFile.readScalarValue(rootGroup, kTransducerSourceFlagName, mTransducerSourceFlag);
300 
301  mInputFile.readScalarValue(rootGroup, kVelocityXSourceFlagName, mVelocityXSourceFlag);
302  mInputFile.readScalarValue(rootGroup, kVelocityYSourceFlagName, mVelocityYSourceFlag);
303  if (isSimulation3D())
304  {
305  mInputFile.readScalarValue(rootGroup, kVelocityZSourceFlagName, mVelocityZSourceFlag);
306  }
307 
308  mInputFile.readScalarValue(rootGroup, kNonUniformGridFlagName, mNonUniformGridFlag);
309  mInputFile.readScalarValue(rootGroup, kAbsorbingFlagName, mAbsorbingFlag);
310  mInputFile.readScalarValue(rootGroup, kNonLinearFlagName, mNonLinearFlag);
311 
312 
313  // Vector sizes.
314  if (mTransducerSourceFlag == 0)
315  {
316  mTransducerSourceInputSize = 0;
317  }
318  else
319  {
320  mTransducerSourceInputSize = mInputFile.getDatasetSize(rootGroup, kTransducerSourceInputName);
321  }
322 
323  // in 2D mVelocityZSourceFlag is always 0
324  if ((mTransducerSourceFlag > 0) || (mVelocityXSourceFlag > 0) ||
325  (mVelocityYSourceFlag > 0) || (mVelocityZSourceFlag > 0))
326  {
327  mVelocitySourceIndexSize = mInputFile.getDatasetSize(rootGroup, kVelocitySourceIndexName);
328  }
329 
330 
331  // uxyz_source_flags
332  if ((mVelocityXSourceFlag > 0) || (mVelocityYSourceFlag > 0) || (mVelocityZSourceFlag > 0))
333  {
334  mInputFile.readScalarValue(rootGroup, kVelocitySourceManyName, mVelocitySourceMany);
335 
336  size_t sourceModeNumericValue = 0;
337  mInputFile.readScalarValue(rootGroup, kVelocitySourceModeName, sourceModeNumericValue);
338 
339  // convert the size_t value to enum
340  switch (sourceModeNumericValue)
341  {
342  case 0: mVelocitySourceMode = SourceMode::kDirichlet;
343  break;
344  case 1: mVelocitySourceMode = SourceMode::kAdditiveNoCorrection;
345  break;
346  case 2: mVelocitySourceMode = SourceMode::kAdditive;
347  break;
348  default:
349  {
350  throw ios::failure(kErrFmtBadVelocitySourceMode);
351  break;
352  }
353  }//case
354  }
355  else
356  {
357  mVelocitySourceMany = 0;
358  mVelocitySourceMode = SourceMode::kDirichlet;
359  }
360 
361  // p_source_flag
362  if (mPressureSourceFlag != 0)
363  {
364  mInputFile.readScalarValue(rootGroup, kPressureSourceManyName, mPressureSourceMany);
365 
366  size_t sourceModeNumericValue = 0;
367  mInputFile.readScalarValue(rootGroup, kPressureSourceModeName, sourceModeNumericValue);
368 
369  // convert the size_t value to enum
370  switch (sourceModeNumericValue)
371  {
372  case 0: mPressureSourceMode = SourceMode::kDirichlet;
373  break;
374  case 1: mPressureSourceMode = SourceMode::kAdditiveNoCorrection;
375  break;
376  case 2: mPressureSourceMode = SourceMode::kAdditive;
377  break;
378  default:
379  {
380  throw ios::failure(kErrFmtBadPressureSourceMode);
381  break;
382  }
383  }//case
384 
385 
386  mPressureSourceIndexSize = mInputFile.getDatasetSize(rootGroup, kPressureSourceIndexName);
387  }
388  else
389  {
390  mPressureSourceMode = SourceMode::kDirichlet;
391  mPressureSourceMany = 0;
392  mPressureSourceIndexSize = 0;
393  }
394 
395 
396  // absorb flag
397  if (mAbsorbingFlag != 0)
398  {
399  mInputFile.readScalarValue(rootGroup, kAlphaPowerName, mAlphaPower);
400  if (mAlphaPower == 1.0f)
401  {
402  throw std::invalid_argument(kErrFmtIllegalAlphaPowerValue);
403  }
404 
405  mAlphaCoeffScalarFlag = mInputFile.getDatasetDimensionSizes(rootGroup, kAlphaCoeffName) == scalarSizes;
406  if (mAlphaCoeffScalarFlag)
407  {
408  mInputFile.readScalarValue(rootGroup, kAlphaCoeffName, mAlphaCoeffScalar);
409  }
410  }
411 
412 
413  mC0ScalarFlag = mInputFile.getDatasetDimensionSizes(rootGroup, kC0Name) == scalarSizes;
414  if (mC0ScalarFlag)
415  {
416  mInputFile.readScalarValue(rootGroup, kC0Name, mC0Scalar);
417  }
418 
419  if (mNonLinearFlag)
420  {
421  mBOnAScalarFlag = mInputFile.getDatasetDimensionSizes(rootGroup, kBonAName) == scalarSizes;
422  if (mBOnAScalarFlag)
423  {
424  mInputFile.readScalarValue(rootGroup, kBonAName, mBOnAScalar);
425  }
426  }
427 
428  mRho0ScalarFlag = mInputFile.getDatasetDimensionSizes(rootGroup, kRho0Name) == scalarSizes;
429  if (mRho0ScalarFlag)
430  {
431  mInputFile.readScalarValue(rootGroup, kRho0Name, mRho0Scalar);
432  mInputFile.readScalarValue(rootGroup, kRho0SgxName, mRho0SgxScalar);
433  mInputFile.readScalarValue(rootGroup, kRho0SgyName, mRho0SgyScalar);
434  if (isSimulation3D())
435  {
436  mInputFile.readScalarValue(rootGroup, kRho0SgzName, mRho0SgzScalar);
437  }
438  }
439 }// end of readScalarsFromInputFile

442 /**
443  * Save scalars into the output HDF5 file.
444  */
445 void Parameters::saveScalarsToOutputFile()
446 {
447  const hid_t rootGroup = mOutputFile.getRootGroup();
448 
449  // Write dimension sizes (Z is always written to distinguish 2D and 3D simulations)
450  mOutputFile.writeScalarValue(rootGroup, kNxName, mFullDimensionSizes.nx);
451  mOutputFile.writeScalarValue(rootGroup, kNyName, mFullDimensionSizes.ny);
452  mOutputFile.writeScalarValue(rootGroup, kNzName, mFullDimensionSizes.nz);
453 
454  mOutputFile.writeScalarValue(rootGroup, kNtName, mNt);
455 
456  mOutputFile.writeScalarValue(rootGroup, kDtName, mDt);
457  mOutputFile.writeScalarValue(rootGroup, kDxName, mDx);
458  mOutputFile.writeScalarValue(rootGroup, kDyName, mDy);
459  if (isSimulation3D())
460  {
461  mOutputFile.writeScalarValue(rootGroup, kDzName, mDz);
462  }
463 
464  mOutputFile.writeScalarValue(rootGroup, kCRefName, mCRef);
465 
466  mOutputFile.writeScalarValue(rootGroup, kPmlXSizeName, mPmlXSize);
467  mOutputFile.writeScalarValue(rootGroup, kPmlYSizeName, mPmlYSize);
468  if (isSimulation3D())
469  {
470  mOutputFile.writeScalarValue(rootGroup, kPmlZSizeName, mPmlZSize);
471  }
472 
473  mOutputFile.writeScalarValue(rootGroup, kPmlXAlphaName, mPmlXAlpha);
474  mOutputFile.writeScalarValue(rootGroup, kPmlYAlphaName, mPmlYAlpha);
475  if (isSimulation3D())
476  {
477  mOutputFile.writeScalarValue(rootGroup, kPmlZAlphaName, mPmlZAlpha);
478  }
479 
480  mOutputFile.writeScalarValue(rootGroup, kPressureSourceFlagName, mPressureSourceFlag);
481  mOutputFile.writeScalarValue(rootGroup, kInitialPressureSourceFlagName, mInitialPressureSourceFlag);
482 
483  mOutputFile.writeScalarValue(rootGroup, kTransducerSourceFlagName, mTransducerSourceFlag);
484 
485  mOutputFile.writeScalarValue(rootGroup, kVelocityXSourceFlagName, mVelocityXSourceFlag);
486  mOutputFile.writeScalarValue(rootGroup, kVelocityYSourceFlagName, mVelocityYSourceFlag);
487  if (isSimulation3D())
488  {
489  mOutputFile.writeScalarValue(rootGroup, kVelocityZSourceFlagName, mVelocityZSourceFlag);
490  }
491 
492  mOutputFile.writeScalarValue(rootGroup, kNonUniformGridFlagName, mNonUniformGridFlag);
493  mOutputFile.writeScalarValue(rootGroup, kAbsorbingFlagName, mAbsorbingFlag);
494  mOutputFile.writeScalarValue(rootGroup, kNonLinearFlagName, mNonLinearFlag);
495 
496 
497 // velocity source flags.
498  if ((mVelocityXSourceFlag > 0) || (mVelocityYSourceFlag > 0) || (mVelocityZSourceFlag > 0))
499  {
500  mOutputFile.writeScalarValue(rootGroup, kVelocitySourceManyName, mVelocitySourceMany);
501  mOutputFile.writeScalarValue(rootGroup, kVelocitySourceModeName, static_cast<size_t>(mVelocitySourceMode));
502  }
503 
504  // pressure source flags.
505  if (mPressureSourceFlag != 0)
506  {
507  mOutputFile.writeScalarValue(rootGroup, kPressureSourceManyName, mPressureSourceMany);
508  mOutputFile.writeScalarValue(rootGroup, kPressureSourceModeName, static_cast<size_t>(mPressureSourceMode));
509  }
510 
511  // absorption flag
512  if (mAbsorbingFlag != 0)
513  {
514  mOutputFile.writeScalarValue(rootGroup, kAlphaPowerName, mAlphaPower);
515  }
516 
517  // if copy sensor mask, then copy the mask type
518  if (getCopySensorMaskFlag())
519  {
520  size_t sensorMaskTypeNumericValue = 0;
521 
522  switch (mSensorMaskType)
523  {
524  case SensorMaskType::kIndex: sensorMaskTypeNumericValue = 0;
525  break;
526  case SensorMaskType::kCorners: sensorMaskTypeNumericValue = 1;
527  break;
528  }// switch
529 
530  mOutputFile.writeScalarValue(rootGroup, kSensorMaskTypeName, sensorMaskTypeNumericValue);
531  }
532 }// end of saveScalarsToFile
